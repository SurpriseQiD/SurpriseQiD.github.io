def handle_outliers(df, columns, method='iqr', threshold=1.5):
    """澶勭悊寮傚父鍊?#34;""
    df_clean = df.copy()
    
    for col in columns:
        if method == 'iqr':
            Q1 = df[col].quantile(0.25)
            Q3 = df[col].quantile(0.75)
            IQR = Q3 - Q1
            lower_bound = Q1 - threshold * IQR
            upper_bound = Q3 + threshold * IQR
            
            # 灏嗗紓甯稿€兼浛鎹负杈圭晫鍊?/span>
            df_clean[col] = df[col].clip(lower_bound, upper_bound)
    
    return df_clean

# 澶勭悊鏁板€肩壒寰佺殑寮傚父鍊?/span>
numeric_features = X.columns.tolist()
X_clean = handle_outliers(X, numeric_features)

def create_features(df):
    """鍒涘缓鏂扮壒寰?#34;""
    df_new = df.copy()
    
    # 鍒涘缓鎴块棿鎬绘暟鐗瑰緛
    df_new['TotalRooms'] = df['AveRooms'] * df['HouseAge']
    
    # 鍒涘缓鎴块棿涓庝汉鍙ｆ瘮渚?/span>
    df_new['RoomsPerHousehold'] = df['AveRooms'] / (df['Population'] / df['Households'] + 1)
    
    # 鍒涘缓鏀跺叆涓庢埧闂存瘮渚?/span>
    df_new['IncomePerRoom'] = df['MedInc'] / (df['AveRooms'] + 1)
    
    return df_new

X_engineered = create_features(X_clean)

models = {
    'Linear Regression': LinearRegression(),
    'Ridge Regression': Ridge(alpha=1.0),
    'Lasso Regression': Lasso(alpha=0.01),
    'Random Forest': RandomForestRegressor(n_estimators=100, random_state=42),
    'Gradient Boosting': GradientBoostingRegressor(n_estimators=100, random_state=42)
}

def evaluate_models(models, X_train, X_test, y_train, y_test):
    """璇勪及澶氫釜妯″瀷"""
    results = {}
    
    for name, model in models.items():
        # 璁粌妯″瀷
        model.fit(X_train, y_train)
        
        # 棰勬祴
        y_pred_train = model.predict(X_train)
        y_pred_test = model.predict(X_test)
        
        # 璁＄畻鎸囨爣
        train_mse = mean_squared_error(y_train, y_pred_train)
        test_mse = mean_squared_error(y_test, y_pred_test)
        train_r2 = r2_score(y_train, y_pred_train)
        test_r2 = r2_score(y_test, y_pred_test)
        
        # 浜ゅ弶楠岃瘉
        cv_scores = cross_val_score(model, X_train, y_train, 
                                   cv=5, scoring='r2')
        
        results[name] = {
            'model': model,
            'train_mse': train_mse,
            'test_mse': test_mse,
            'train_r2': train_r2,
            'test_r2': test_r2,
            'cv_mean': cv_scores.mean(),
            'cv_std': cv_scores.std()
        }
        
        print(f"\n{name}:")
        print(f"  璁粌闆?R虏: {train_r2:.4f}")
        print(f"  娴嬭瘯闆?R虏: {test_r2:.4f}")
        print(f"  浜ゅ弶楠岃瘉 R虏: {cv_scores.mean():.4f} (卤{cv_scores.std():.4f})")
    
    return results

# 璇勪及妯″瀷
results = evaluate_models(models, X_train_scaled, X_test_scaled, y_train, y_test)

from sklearn.model_selection import GridSearchCV

def tune_random_forest(X_train, y_train):
    """闅忔満妫灄瓒呭弬鏁拌皟浼?#34;""
    param_grid = {
        'n_estimators': [100, 200, 300],
        'max_depth': [10, 20, 30, None],
        'min_samples_split': [2, 5, 10],
        'min_samples_leaf': [1, 2, 4],
        'max_features': ['sqrt', 'log2']
    }
    
    rf = RandomForestRegressor(random_state=42)
    grid_search = GridSearchCV(
        rf, param_grid, cv=5, scoring='r2',
        n_jobs=-1, verbose=1
    )
    
    grid_search.fit(X_train, y_train)
    
    print(f"鏈€浣冲弬鏁? {grid_search.best_params_}")
    print(f"鏈€浣充氦鍙夐獙璇佸緱鍒? {grid_search.best_score_:.4f}")
    
    return grid_search.best_estimator_

def save_model_pipeline(model, scaler, feature_names, model_name='best_model'):
    """淇濆瓨妯″瀷鍜岄澶勭悊绠￠亾"""
    import joblib
    import datetime
    
    # 鍒涘缓妯″瀷淇℃伅瀛楀吀
    model_info = {
        'model': model,
        'scaler': scaler,
        'feature_names': feature_names,
        'created_at': datetime.datetime.now().strftime("%Y-%m-%d %H:%M:%S"),
        'version': '1.0'
    }
    
    # 淇濆瓨妯″瀷
    joblib.dump(model_info, f'{model_name}.pkl')
    print(f"妯″瀷宸蹭繚瀛樹负 {model_name}.pkl")
    
    # 淇濆瓨妯″瀷鍏冩暟鎹?/span>
    metadata = {
        'model_type': type(model).__name__,
        'features': feature_names,
        'created_at': model_info['created_at'],
        'performance': results[best_model_name] if best_model_name in results else {}
    }
    
    import json
    with open(f'{model_name}_metadata.json', 'w') as f:
        json.dump(metadata, f, indent=2)
    
    return model_info

# 閫夋嫨鏈€浣虫ā鍨嬩繚瀛?/span>

    def __init__(self, model_path='housing_price_predictor.pkl'):

        self.model_info = joblib.load(model_path)
        self.model = self.model_info['model']
        self.scaler = self.model_info['scaler']
        self.feature_names = self.model_info['feature_names']

        """棰勫鐞嗚緭鍏ユ暟鎹?#34;""
        # 纭繚杈撳叆涓篋ataFrame
        if isinstance(input_data, dict):

        elif isinstance(input_data, pd.DataFrame):
            df = input_data.copy()
        else:
            raise ValueError("杈撳叆搴斾负瀛楀吀鎴朌ataFrame")
        
        # 纭繚鍒楅『搴忔纭?
